function loginUser(){
var u_name = $("#username").val().trim();
var u_pwd = $("#password").val().trim();
if(u_name==""||u_pwd==""){
$('.errormessege').html("用户名或密码不能为空");
return false;
}else{
if(u_name.length>100||u_pwd.length>100){
$('.errormessege').html('用户名或密码长度超出范围');
return false;
}
if($("#oldTest").is(":visible")){//旧的验证方式
var code = $("#veryCode").val();
if(code.length!=4&&code.length!=0){
$('.errormessege').html('验证码只能是四位数字');
return false;
}else if(code.length==0){
$('.errormessege').html('验证码不能为空');
return false;
}
$.ajax({
type:"POST",
url:"/hbaq/resultServlet.do",
data:{"c":code},
success:function(data){
var json = eval('(' + data + ')');
if(json.status == "100"){
document.all.loginForm.submit();
}else{
changeImg();
if(json.status == "101"){
$('.errormessege').html('验证码异常');
}else if(json.status == "102"){
$('.errormessege').html('验证码超时');
}else if(json.status == "103"){
$('.errormessege').html('验证码错误');
}
return false;
}
}
});
}else{//急验验证
geetest_ajax_results();
}
}
}
function EnterPress(event){
var e = event || window.event || arguments.callee.caller.arguments[0];
if(e.keyCode == 13){
loginUser();
}
}

}

var gtFailbackFrontInitial = function(result) {
var s = document.createElement('script');
s.id = 'gt_lib';
s.src = 'http://static.geetest.com/static/js/geetest.0.0.0.js';
s.charset = 'UTF-8';
s.type = 'text/javascript';
document.getElementsByTagName('head')[0].appendChild(s);
var loaded = false;
s.onload = s.onreadystatechange = function() {
if (!loaded
&& (!this.readyState
|| this.readyState === 'loaded' || this.readyState === 'complete')) {
loadGeetest(result);
loaded = true;
}
};
};
//get geetest server status, use the failback solution
var loadGeetest = function(config) {
//1. use geetest captcha
window.gt_captcha_obj = new window.Geetest({
gt : config.gt,
challenge : config.challenge,
product : 'float',
//product : 'embed',
offline : !config.success
});
gt_captcha_obj.appendTo("#div_id_embed");
/* //Ajax request demo,if you use submit form ,then ignore it
gt_captcha_obj.onSuccess(function() {
//geetest_ajax_results();
loginUser();
}); */
};
s = document.createElement('script');
s.src = 'http://api.geetest.com/get.php?callback=gtcallback';
$("#div_geetest_lib").append(s);
var gtcallback =( function() {
var status = 0, result, apiFail;
return function(r) {
status += 1;
if (r) {
result = r;
setTimeout(function() {
if (!window.Geetest) {
apiFail = true;
gtFailbackFrontInitial(result);
}
}, 1000);
}
else if(apiFail) {
return
}
if (status == 2) {
loadGeetest(result);
}
};
})();
